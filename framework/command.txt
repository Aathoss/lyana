package framework

type (
    Command struct {
        Name       string
        Permission int
        Aliases    []string
        Help       string
        Debug      bool
        Module     string
        Run        func(ctx Context)
    }

    Commands map[string]Command
)

// Permet d'enregistrer une nouvelle commande
func (commands Commands) Register(command Command) {
    commands[command.Name] = command
}

// Permet d'ajouter un tableau de commandes
func (commands Commands) RegisterMultiple(multiple []Command) {
    for _, command := range multiple {
        commands[command.Name] = command
    }
}

// Récupère toutes les commandes
func (commands Commands) GetAll() Commands {
    return commands
}

// Récupère une commande par son nom
func (commands Commands) Get(name string) Command {
    return commands[name]
}

// Vérifie les permissions d'une commande
func (commands Commands) CheckPermission(command Command, permission int) bool {
    if permission >= command.Permission {
        return true
    }

    return false
}
